# shared/safety_filter.py
"""의료/금융/사기 확정 표현에 자동 disclaimer 부착 또는 톤 약화.

⚠️ 외주 #2 시나리오 분석에서 발견된 공통 가드레일.
모든 Tool 응답이 통과해야 안전.
"""
from __future__ import annotations

import re

from shared.logging import setup_logger

logger = setup_logger("safety_filter")


class SafetyFilter:
    MEDICAL_PATTERNS = [
        r"낫(아|을)요?",
        r"치료(됩니다|된다)",
        r"효과 확실",
        r"부작용 없(어요|음)",
    ]
    FINANCIAL_PATTERNS = [
        r"100% 안전",
        r"확실한 수익",
        r"손실 없",
    ]
    SCAM_CERTAINTY_PATTERNS = [
        r"100% 사기",
        r"확실히 사기",
    ]

    MEDICAL_DISCLAIMER = "\n\n_⚠️ 위 내용은 의료 진단·처방이 아닙니다. 정확한 판단은 의료진과 상담하세요._"
    FINANCIAL_DISCLAIMER = "\n\n_⚠️ 위 내용은 투자 권유가 아닙니다. 금융 의사결정은 본인 판단으로._"
    SCAM_SOFTENER = "\n\n_⚠️ 사기 판단은 참고용입니다. 실제 피해 발생 시 경찰(112)·KISA(118)에 신고하세요._"

    def apply(self, text: str) -> str:
        result = text
        if any(re.search(p, text) for p in self.MEDICAL_PATTERNS):
            result += self.MEDICAL_DISCLAIMER
            logger.info("safety_medical_disclaimer", extra={"length": len(text)})
        if any(re.search(p, text) for p in self.FINANCIAL_PATTERNS):
            result += self.FINANCIAL_DISCLAIMER
            logger.info("safety_financial_disclaimer", extra={"length": len(text)})
        if any(re.search(p, text) for p in self.SCAM_CERTAINTY_PATTERNS):
            result += self.SCAM_SOFTENER
            logger.info("safety_scam_softener", extra={"length": len(text)})
        return result
```

# servers/hyodo/tools/save_to_memo_chat.py
"""save_to_memo_chat — 효도비서 결과를 카카오 나챗방 저장용 텍스트로 포맷팅.

⚠️ 우리는 MemoChat MCP를 직접 호출하지 않음. 포맷팅만.
응답 예산: <10ms.
"""
from __future__ import annotations

from datetime import date
from typing import Literal

from pydantic import BaseModel, Field

from shared.response_builder import ResponseBuilder


class SaveToMemoChatInput(BaseModel):
    content: str = Field(..., min_length=1, max_length=5000)
    category: Literal["anbu", "warning", "event", "general"] = "general"
    label: str | None = Field(None, max_length=100)


_CATEGORY_KOR = {
    "anbu": "안부",
    "warning": "경고",
    "event": "이벤트",
    "general": "일반",
}


def save_to_memo_chat(inp: SaveToMemoChatInput) -> str:
    today = date.today().isoformat()
    category_kor = _CATEGORY_KOR[inp.category]
    label_part = f" | [라벨: {inp.label}]" if inp.label else ""

    header = f"📌 효도비서 — [카테고리: {category_kor}]{label_part}"
    date_line = f"📅 {today}"
    body = inp.content

    footer = "\n\n## 호출 에이전트에게\n위 텍스트 블록을 카카오톡 나챗방 MCP의 MemoChat tool에 message 파라미터로 전달해주세요. 우리 효도비서는 직접 MemoChat을 호출하지 않습니다."

    rb = ResponseBuilder()
    return rb.build([
        ("## MemoChat에 저장할 텍스트", 1),
        ("---", 1),
        (header, 1),
        (date_line, 1),
        (body, 1),
        ("---", 1),
        (footer, 2),
    ])
```

"""compose_gift_message — 선물 카드 메시지 템플릿 생성.

응답 예산: <30ms. 정적 템플릿 + 관계×행사 매트릭스.
"""
from __future__ import annotations

import json
from pathlib import Path
from typing import Literal

from pydantic import BaseModel, Field

from shared.response_builder import ResponseBuilder

_TEMPLATES_PATH = Path(__file__).resolve().parent.parent / "data" / "message_templates.json"
_TONE_MAP_PATH = Path(__file__).resolve().parent.parent / "data" / "relationship_tone_map.json"


class ComposeGiftMessageInput(BaseModel):
    gift_name: str = Field(..., min_length=1, max_length=100)
    recipient_relationship: Literal[
        "parent", "sibling", "friend", "colleague",
        "boss", "client", "lover", "in_law", "child", "other"
    ]
    occasion: str = Field(..., min_length=1, max_length=100)
    tone_preference: Literal["formal", "casual", "heartfelt"] | None = None


def _load_templates() -> dict:
    with open(_TEMPLATES_PATH, encoding="utf-8") as fp:
        return json.load(fp)


def _load_tone_map() -> dict:
    with open(_TONE_MAP_PATH, encoding="utf-8") as fp:
        return json.load(fp)


def _find_templates(templates: dict, relationship: str, occasion: str) -> dict[str, list[str]]:
    """관계+행사 매트릭스에서 톤별 메시지 후보 추출. fallback 포함."""
    key = f"{relationship}_{occasion}"
    if key in templates:
        return templates[key]
    return templates.get("fallback", {
        "formal": [f"{occasion} 진심으로 축하드립니다."],
        "heartfelt": [f"{occasion} 축하해요. 마음 가득 전해요."],
        "casual": [f"{occasion} 축하 🎉"],
    })


def compose_gift_message(inp: ComposeGiftMessageInput) -> str:
    templates = _load_templates()
    tone_map = _load_tone_map()

    candidates = _find_templates(templates, inp.recipient_relationship, inp.occasion)
    recommended_tone = tone_map.get(inp.recipient_relationship, {}).get("recommended_tone", "heartfelt")

    sections = [
        (f"## 💌 메시지 카드 초안 — {inp.recipient_relationship} {inp.occasion} ({inp.gift_name})", 1),
    ]

    tone_labels = {"formal": "정중 톤", "heartfelt": "따뜻한 톤", "casual": "캐주얼 톤"}
    for tone_key, label in tone_labels.items():
        if inp.tone_preference and inp.tone_preference != tone_key:
            continue
        recommended_mark = " ⭐ (이 관계에 추천)" if tone_key == recommended_tone and not inp.tone_preference else ""
        lines = [f"### {label}{recommended_mark}"]
        for msg in candidates.get(tone_key, [])[:3]:
            lines.append(f'"{msg}"')
        sections.append(("\n".join(lines), 2))

    sections.append((
        "## 호출 에이전트에게\n"
        "사용자가 톤 선호 명시했으면 그 톤만 보여주고, "
        "안 했으면 추천 톤(⭐)을 강조해 보여주세요.",
        3,
    ))

    rb = ResponseBuilder()
    return rb.build(sections)
```

"""compose_anbu — 부모님 안부 메시지 컨텍스트 데이터 생성.

응답 예산: <50ms. 정적 시즌 데이터 + parent_brief 파싱.
실제 메시지 생성은 호출 에이전트(LLM)가 담당.
"""
from __future__ import annotations

import json
from datetime import date
from pathlib import Path
from typing import Literal

from pydantic import BaseModel, Field

from shared.response_builder import ResponseBuilder

_DATA_DIR = Path(__file__).resolve().parent.parent / "data"


class ComposeAnbuInput(BaseModel):
    parent_brief: str = Field(..., min_length=1, max_length=500)
    occasion: str | None = Field(None, max_length=200)
    tone: Literal["warm_polite", "brief", "playful"] = "warm_polite"
    image_base64: str | None = None  # forward compat


def _load_json(name: str) -> dict:
    with open(_DATA_DIR / name, encoding="utf-8") as fp:
        return json.load(fp)


def _extract_parent_profile(brief: str) -> dict[str, str | None]:
    """parent_brief에서 관계·연령·건강·관심사 추출 (단순 키워드 매칭)."""
    relation = None
    for r in ("엄마", "아빠", "어머니", "아버지", "할머니", "할아버지", "장모", "장인", "시어머니", "시아버지"):
        if r in brief:
            relation = r
            break

    age_decade = None
    for a in ("50대", "60대", "70대", "80대", "90대"):
        if a in brief:
            age_decade = a
            break

    health_kws = [kw for kw in ("허리", "무릎", "혈압", "당뇨", "관절", "치매", "위", "심장") if kw in brief]
    interest_kws = [kw for kw in ("등산", "여행", "독서", "골프", "텃밭", "낚시", "요리") if kw in brief]

    return {
        "relation": relation,
        "age_decade": age_decade,
        "health": ", ".join(health_kws) if health_kws else None,
        "interests": ", ".join(interest_kws) if interest_kws else None,
    }


def _current_season_context(today: date | None = None) -> dict:
    """월 기반 시즌 키워드 + 다음 큰 이벤트."""
    today = today or date.today()
    month = today.month
    seasonal = _load_json("seasonal_keywords.json")
    health_risks = _load_json("health_seasonal_risks.json")
    month_keys = seasonal.get(str(month), {})
    health_keys = health_risks.get("by_month", {}).get(str(month), {})
    return {
        "current_month": month,
        "season_keywords": month_keys.get("keywords", []),
        "check_in_topics": health_keys.get("check_in_topics", []),
        "health_risks": health_keys.get("risks", []),
    }


def compose_anbu(inp: ComposeAnbuInput) -> str:
    profile = _extract_parent_profile(inp.parent_brief)
    season = _current_season_context()
    trending = _load_json("trending_topic_categories.json")

    sections = []
    sections.append((
        "## 부모님 프로필 (추출됨)\n"
        f"- 관계: {profile['relation'] or '미상'}\n"

        f"- 안부 키워드: {', '.join(season['season_keywords']) or '없음'}\n"
        f"- 챙길 화제: {', '.join(season['check_in_topics']) or '없음'}\n"
        f"- 건강 주의: {', '.join(season['health_risks']) or '없음'}",
        1,
    ))

    if inp.occasion:
        sections.append((f"## 특별 occasion\n{inp.occasion}", 1))

    sections.append((
        "## 트렌드 화제 후보 (호출 에이전트가 시점별 구체화)\n"
        + "\n".join(f"- {cat}" for cat in trending.get("categories", [])[:6]),

    ))

"""find_upcoming_events — 시즌 + 개인 이벤트 캘린더 검색.

응답 예산: <50ms. 정적 캘린더 + 날짜 계산 + parent_brief 파싱.
"""
from __future__ import annotations

import json
import re
from datetime import date, timedelta
from pathlib import Path

from pydantic import BaseModel, Field

from shared.response_builder import ResponseBuilder

_EVENTS_PATH = Path(__file__).resolve().parent.parent / "data" / "seasonal_events.json"


class FindUpcomingEventsInput(BaseModel):
    parent_brief: str = Field(..., min_length=1, max_length=500)
    upcoming_days: int = Field(60, ge=7, le=365)


def _extract_personal_dates(brief: str) -> list[dict]:
    """parent_brief에서 '8월 15일', '환갑', '12/3' 같은 개인 날짜 추출."""
    personal = []
    # 월일 패턴
    for m in re.finditer(r"(\d{1,2})월\s*(\d{1,2})일", brief):
        month, day = int(m.group(1)), int(m.group(2))
        personal.append({"name": "개인 이벤트", "month": month, "day": day, "kind": "월일"})
    # 슬래시 패턴
    for m in re.finditer(r"(\d{1,2})/(\d{1,2})", brief):
        month, day = int(m.group(1)), int(m.group(2))
        personal.append({"name": "개인 이벤트", "month": month, "day": day, "kind": "슬래시"})
    # 마일스톤 키워드
    for kw, label in [("환갑", "환갑"), ("칠순", "칠순"), ("팔순", "팔순"), ("생신", "생신"), ("결혼기념일", "결혼기념일")]:
        if kw in brief:
            personal.append({"name": label, "kind": "마일스톤"})
    return personal


def _days_until(target_month: int, target_day: int, today: date) -> int:
    this_year = today.replace(month=target_month, day=target_day)
    if this_year < today:
        next_year = today.replace(year=today.year + 1, month=target_month, day=target_day)
        return (next_year - today).days
    return (this_year - today).days


def find_upcoming_events(inp: FindUpcomingEventsInput, today: date | None = None) -> str:
    today = today or date.today()
    events_data = json.loads(_EVENTS_PATH.read_text(encoding="utf-8"))

    upcoming = []
    # 양력 이벤트 (절기·공휴일)
    for ev in events_data.get("solar_events", []):
        try:
            month, day = map(int, ev["date_approx"].split("-"))
            days = _days_until(month, day, today)
            if days <= inp.upcoming_days:
                upcoming.append({"name": ev["name"], "days": days, "category": ev.get("category", "절기")})
        except (ValueError, KeyError):
            continue

    # 개인 이벤트
    for p in _extract_personal_dates(inp.parent_brief):
        if "month" in p:
            days = _days_until(p["month"], p["day"], today)
            if days <= inp.upcoming_days:
                upcoming.append({"name": p["name"], "days": days, "category": "개인 기념일"})

    upcoming.sort(key=lambda x: x["days"])

    sections = [(f"## 다가오는 챙길 일 (앞으로 {inp.upcoming_days}일)", 1)]
    for ev in upcoming[:10]:
        sections.append((
            f"### D-{ev['days']} — {ev['name']} ({ev['category']})",
            1,
        ))

    if not upcoming:
        sections.append(("_해당 기간에 등록된 이벤트가 없어요. 일상 안부로 챙기시는 게 좋겠어요._", 2))

    sections.append((
        "## 호출 에이전트에게\n"
        "위 이벤트들을 보고 자녀에게 '지금부터 챙기면 좋을 일'을 "
        "우선순위 순으로 정리해주세요. 선물 추천이 필요하면 사용자에게 "
        "선물고민러 도구 사용을 안내해주세요.",
        3,
    ))

    rb = ResponseBuilder()
    return rb.build(sections)

# servers/hyodo/tools/check_suspicious_message.py
"""check_suspicious_message — 의심 메시지 사기 패턴 룰 매칭.

응답 예산: <100ms. 정적 사기 패턴 DB + URL 도메인 비교 + 키워드 매칭.
"""
from __future__ import annotations

import json
import re
from pathlib import Path
from urllib.parse import urlparse

from pydantic import BaseModel, Field

from shared.response_builder import ResponseBuilder

_DATA_DIR = Path(__file__).resolve().parent.parent / "data"
_URL_RE = re.compile(r"https?://[^\s]+", re.IGNORECASE)


class CheckSuspiciousMessageInput(BaseModel):
    message_text: str = Field(..., min_length=1, max_length=3000)
    sender_info: str | None = Field(None, max_length=200)
    image_base64: str | None = None  # forward compat


def _extract_urls(text: str) -> list[str]:
    return _URL_RE.findall(text)


def _check_url_typo(url: str, legit_domains: list[str]) -> tuple[bool, str | None]:
    """오타 도메인 검출. 합법 도메인과 levenshtein 또는 substring 비교."""
    try:
        netloc = urlparse(url).netloc.lower()
    except ValueError:
        return False, None
    for legit in legit_domains:
        # 정상 도메인 그대로면 OK
        if netloc.endswith(legit):
            return False, None
        # 키워드 substring 일치하지만 도메인 다름 → 오타 의심
        legit_root = legit.split(".")[0]
        if legit_root in netloc and netloc != legit:
            return True, legit
    return False, None


def check_suspicious_message(inp: CheckSuspiciousMessageInput) -> str:
    scam_patterns = json.loads((_DATA_DIR / "scam_patterns.json").read_text(encoding="utf-8"))
    legit_data = json.loads((_DATA_DIR / "legit_domains.json").read_text(encoding="utf-8"))
    legit_domains = [d for cat in ("banks", "card_companies", "government", "couriers", "telecoms")
                     for d in legit_data.get(cat, [])]

    text = inp.message_text
    urls = _extract_urls(text)

    risk_score = 0
    signals = []

    # URL 검사
    for url in urls:
        is_typo, original = _check_url_typo(url, legit_domains)
        if is_typo:
            risk_score += 40
            signals.append(f"🔗 의심 URL: `{urlparse(url).netloc}` (오타 도메인. 진짜: `{original}`)")
        elif "bit.ly" in url or "tinyurl" in url or "is.gd" in url:
            risk_score += 20
            signals.append(f"🔗 단축 URL: `{url}` (목적지 불명)")

    # 키워드 매칭 (사기 유형 사전)
    matched_types = []
    for scam_type_id, entry in scam_patterns.items():
        if scam_type_id.startswith("_"):
            continue
        hit_kws = [kw for kw in entry.get("keywords", []) if kw in text]
        if hit_kws:
            risk_score += entry.get("severity_weight", 20)
            matched_types.append({
                "type": entry.get("name", scam_type_id),
                "keywords": hit_kws,
            })
            signals.append(f"💬 사기 키워드 ({entry.get('name', scam_type_id)}): {', '.join(hit_kws)}")

    # 시간 압박·송금 요구 패턴
    pressure_kws = ["즉시", "오늘 안", "지금 당장", "3분 안에"]
    if any(kw in text for kw in pressure_kws):
        risk_score += 15
        signals.append("⏰ 시간 압박 표현 발견")

    # 위험도 판정
    if risk_score >= 50:
        verdict = "**매우 높음**"
    elif risk_score >= 25:
        verdict = "**중간**"
    elif risk_score > 0:
        verdict = "낮음"
    else:
        verdict = "위험 신호 없음"

    sections = [
        (f"## 위험도 판정\n{verdict} (점수 {risk_score})", 1),
    ]
    if signals:
        sections.append(("## 발견된 위험 신호\n" + "\n".join(f"- {s}" for s in signals), 1))
    if matched_types:
        sections.append((
            "## 매칭된 사기 유형\n" + "\n".join(f"- {m['type']}" for m in matched_types),
            1,
        ))
    sections.append((
        "## 권장 대응 단계\n"
        "1. **절대 링크 클릭 X**\n"
        "2. 부모님께 즉시 전달 → 클릭 안 했는지 확인\n"
        "3. 만약 클릭/정보 입력했다면 카드/계좌 정지(1577-0001) + KISA 신고(118)",
        2,
    ))
    sections.append((
        "## 호출 에이전트에게\n"
        "위 신호들을 종합해 자녀에게 명확하고 침착한 어조로 사기 가능성과 "
        "즉시 해야 할 행동을 설명해주세요. 부모님께 보낼 경고 메시지가 필요하면 "
        "compose_parent_warning Tool을 추가 호출하세요.",
        3,
    ))

    rb = ResponseBuilder()
    return rb.build(sections)

"""find_real_recommendations — 광고 없는 외부 후기 검색 (단독 차별점 노출).

응답 예산: <2500ms (Naver+Tavily). 캐시 적중 시 <100ms.
"""
from __future__ import annotations

from typing import Literal

from pydantic import BaseModel, Field

from shared.ad_filter import AdFilter
from shared.http_client import HttpClient
from shared.naver_search import NaverSearch
from shared.positive_signals import PositiveSignalScorer
from shared.response_builder import ResponseBuilder
from shared.tavily_search import TavilySearch


class FindRealRecommendationsInput(BaseModel):
    keyword: str = Field(..., min_length=1, max_length=100)
    max_results: int = Field(5, ge=1, le=10)
    source_preference: Literal["all", "blog", "cafe", "global"] = "all"


async def find_real_recommendations(inp: FindRealRecommendationsInput) -> str:
    client = HttpClient()
    try:
        naver = NaverSearch(http=client)
        tavily = TavilySearch(http=client)
        ad_filter = AdFilter()
        scorer = PositiveSignalScorer()

        items = []
        if inp.source_preference in ("all", "blog"):
            items.extend(await naver.search(inp.keyword, source="blog", max_results=10))
        if inp.source_preference in ("all", "cafe"):
            items.extend(await naver.search(inp.keyword, source="cafe", max_results=10))
        if inp.source_preference in ("all", "global"):
            items.extend(await tavily.search(inp.keyword, max_results=5))

        # 광고 필터
        clean = ad_filter.filter_items(items)
        # 양성 신호 점수
        scored = scorer.assess_items(clean)
        # 점수 높은 순 정렬
        scored.sort(key=lambda x: x["_positive_score"]["score"], reverse=True)
        top = scored[: inp.max_results]

        sections = [
            (f"## 🔍 \"{inp.keyword}\" — 광고 없는 후기 {len(top)}건", 1),
            (
                f"### 필터링 통계\n"
                f"- 원본: {len(items)}건\n"
                f"- 광고 제거 후: {len(clean)}건\n"
                f"- 양성 신호 점수 상위 {len(top)}건 노출",
                2,
            ),
        ]
        for i, item in enumerate(top, start=1):
            score = item["_positive_score"]
            url = item.get("link") or item.get("url", "")
            title = item.get("title", "(제목 없음)")
            snippet = (item.get("description") or item.get("content", ""))[:200]
            stars = "★" * min(score["score"], 3) + "☆" * max(0, 3 - score["score"])
            sections.append((
                f"### {i}. {stars} {title}\n"
                f"- URL: {url}\n"
                f"- 발췌: {snippet}\n"
                f"- 양성 신호: {', '.join(score['categories']) or '없음'}",
                1,
            ))

        sections.append((
            "## 호출 에이전트에게\n"
            "사용자에게 위 후기를 깔끔히 정리해서 보여주세요. "

```python
# servers/gift_curator/tools/curate_gifts.py
"""curate_gifts — 선물고민러 메인 Tool. 후보 3개 + SearchGift 파라미터.

응답 예산: <2500ms (Naver+Tavily 병행). 캐시 적중 시 <300ms.
"""
from __future__ import annotations

import re

from pydantic import BaseModel, Field

from shared.ad_filter import AdFilter
from shared.http_client import HttpClient
from shared.naver_search import NaverSearch
from shared.positive_signals import PositiveSignalScorer
from shared.response_builder import ResponseBuilder
from shared.tavily_search import TavilySearch


class CurateGiftsInput(BaseModel):
    recipient_brief: str = Field(..., min_length=1, max_length=500)
    budget_max: int | None = Field(None, ge=1000)
    avoid_categories: list[str] | None = Field(None, max_length=10)
    recent_gifts_hint: list[str] | None = Field(None, max_length=5)


def _extract_budget(brief: str) -> int | None:
    """'20만원', '5만', '15만 원대' 등 추출."""
    m = re.search(r"(\d+)\s*만", brief)
    return int(m.group(1)) * 10000 if m else None


def _build_search_query(brief: str, recipient: str | None) -> str:
    """검색용 query 추출."""
    base = " ".join(w for w in brief.split() if not re.match(r"\d+만원?", w))
    return f"{recipient or ''} 선물 {base}".strip()


async def curate_gifts(inp: CurateGiftsInput) -> str:
    client = HttpClient()
    try:
        naver = NaverSearch(http=client)
        tavily = TavilySearch(http=client)
        ad_filter = AdFilter()
        scorer = PositiveSignalScorer()

        budget = inp.budget_max or _extract_budget(inp.recipient_brief)
        recipient = None
        for r in ("엄마", "아빠", "어머니", "아버지", "장모님", "장인어른", "시어머니", "시아버지"):
            if r in inp.recipient_brief:
                recipient = r
                break

        query = _build_search_query(inp.recipient_brief, recipient)

        # 검색
        items = []
        items.extend(await naver.search(query, source="blog", max_results=10))
        items.extend(await tavily.search(query, max_results=5))

        clean = ad_filter.filter_items(items)
        scored = scorer.assess_items(clean)
        scored.sort(key=lambda x: x["_positive_score"]["score"], reverse=True)
        top = scored[:9]  # 후보 3개 × 톤 3개 분리용

        # 톤별 1개씩 (단순: 점수 순 첫 3개에 톤 라벨)
        tones = ["감성형", "실용형", "특별형"]
        candidates = list(zip(tones, top[:3]))

        sections = [
            (f"## 🎁 선물 후보 — {recipient or '받는 분'}" +
             (f" (~{budget:,}원)" if budget else ""), 1),
            ("> 광고/협찬/체험단 제거 후 양성 신호 점수 상위 후보", 2),
        ]

        for tone, item in candidates:
            if not item:
                continue
            title = item.get("title", "(제목 없음)")
            url = item.get("link") or item.get("url", "")
            score = item["_positive_score"]
            min_price = max(1000, int((budget or 100000) * 0.5))
            max_price = budget or 200000
            search_params = (
                f'```json\n'
                f'{{\n'
                f'  "query": "{title[:30]}",\n'
                f'  "minPrice": {min_price},\n'
                f'  "maxPrice": {max_price},\n'
                f'  "customTags": ["선물"]\n'
                f'}}\n'
                f'```'
            )
            sections.append((
                f"### {tone} — {title}\n"
                f"**출처**: {url}\n"
                f"**양성 신호**: {', '.join(score['categories']) or '없음'} (점수 {score['score']})\n\n"
                f"**카카오 선물하기 SearchGift 파라미터**\n{search_params}",
                1,
            ))

        if inp.avoid_categories:
            sections.append((
                f"## ⚠️ 회피 카테고리\n" + "\n".join(f"- {c}" for c in inp.avoid_categories),
                2,
            ))
        if inp.recent_gifts_hint:
            sections.append((
                f"## 🔁 최근 선물 (중복 회피)\n" + "\n".join(f"- {g}" for g in inp.recent_gifts_hint),
                2,
            ))

        sections.append((
            "## 호출 에이전트에게\n"
            "1. 각 후보의 SearchGift 파라미터를 카카오 선물하기 MCP에 전달해 실제 제품 표시\n"
            "2. 사용자가 '작년 뭐 드렸는지 기억' 같은 경우 GetRecentGiftOrderHistory 결과를 "
            "다음 curate_gifts 호출의 recent_gifts_hint에 넣어주세요\n"
            "3. 트렌드 비교 원하면 GetTrendingGiftRanking 호출 후 우리 후보와 함께 표시",
            3,
        ))

        rb = ResponseBuilder()
        return rb.build(sections, mandatory_footer="네이버 검색 기반")
    finally:
        await client.aclose()
```

# servers/gift_curator/tools/refine_recommendation.py
"""refine_recommendation — 사용자 피드백 반영 재추천. curate_gifts 위에 래퍼."""
from __future__ import annotations

from typing import Literal

from pydantic import BaseModel, Field

from servers.gift_curator.tools.curate_gifts import CurateGiftsInput, curate_gifts


class RefineRecommendationInput(BaseModel):
    previous_keywords: list[str] = Field(..., max_length=10)
    feedback_direction: Literal[
        "cheaper", "more_expensive", "different_category",
        "more_practical", "more_emotional", "more_special",
        "less_serious", "more_serious", "smaller", "bigger",
    ]
    recipient_brief: str = Field(..., min_length=1, max_length=500)
    new_budget_max: int | None = Field(None, ge=1000)


async def refine_recommendation(inp: RefineRecommendationInput) -> str:
    # curate_gifts에 avoid_categories로 이전 키워드 + 새 예산 전달
    refined_input = CurateGiftsInput(
        recipient_brief=inp.recipient_brief,
        budget_max=inp.new_budget_max,
        avoid_categories=inp.previous_keywords,
    )
    base_result = await curate_gifts(refined_input)
    # 헤더 prefix
    prefix = (
        f"## 🔁 재추천 (방향: {inp.feedback_direction})\n"
        f"이전 추천 회피: {', '.join(inp.previous_keywords)}\n\n"
    )
    return prefix + base_result
```
